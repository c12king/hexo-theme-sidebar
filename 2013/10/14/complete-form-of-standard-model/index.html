Complete form of the standard model

\begin{equation} \mathcal{L} = \mathcal{L}{gauge} + \mathcal{L}{f} + \mathcal{L}{\phi} + \mathcal{L}{Yuk} \end{equation}

\begin{equation} \mathcal{L}{gauge} = - \frac{1}{4} G^a{\mu \nu} Ga^{\mu \nu} - \frac{1}{4} G^i{\mu \nu} G_i^{\mu \nu} 

- \frac{1}{4} B_{\mu \nu} B^{\mu \nu}

\end{equation}

\begin{eqnarray} \mathcal{L}f = \sum{m =1}^F \left( \overline{q}^0{mL} i \gamma^{\mu}D{\mu} q^0_{mL}

+ \overline{l}^0_{mL} i \gamma^{\mu}D_{\mu} l^0_{mL} 
+ \overline{u}^0_{mR} i \gamma^{\mu}D_{\mu} u^0_{mR}  \\\
+ \overline{d}^0_{mR} i \gamma^{\mu}D_{\mu} u^0_{mR}
+ \overline{e}^0_{mR} i \gamma^{\mu}D_{\mu} e^0_{mR}
+ \overline{\nu}^0_{mR} i \gamma^{\mu}D_{\mu} \nu^0_{mR} \right)

\end{eqnarray}
